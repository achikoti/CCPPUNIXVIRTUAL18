4. Write a C program to check whether year is leap year or not using conditional operator.

sol)
#include<stdio.h>

int main() {
   int y,x;
   printf("Enter the year:");
   scanf("%d",&y);
   x=(y%400 == 0)?:(y%100 != 0)?(y%4 == 0)?1:0:0;
   if(x==1)
   {
       printf("\n");
       printf("%d is a leap year",y);
   }
   else
   {
       printf("\n");
       printf("%d is not a leap year",y);
   }
}

Test Data:
Enter the year:
2007
Output:
2007 is not a leap year

5. Write a C program to print all Armstrong numbers between 1 to n.
 
sol)
#include<stdio.h>
int main()
{
int i, a, r, s, n ;
printf("Enter the limit : ") ;
scanf("%d",&n) ;
printf("\nThe armstrong numbers from 1 to %d are:\n\n",n) ;
for(i = 0 ; i <= n ; i++)
{
a = i ;
s = 0 ;
while(a > 0)
{
r = a % 10 ;
s = s + (r * r * r) ;
a = a / 10 ;
}
if(i == s)
printf("%d\t", i) ;
}
}

7. C program for ternary operator?

sol)
int main() {
   int x,y,z;
   printf("Enter two values:");
   scanf("%d",&x);
   scanf("%d",&y);
   z=(x>y)?1:0;
   if(z==1)
   {
       printf("\n");
       printf("%d is greater",x);
   }
   else
   {
       printf("\n");
       printf("%d is greater",y);
   }
}

Test Data:
Enter two values:
1
2
Output:
2 is greater
