sol)

5. C program to transfer data from one file to another file.

sol)
#include<stdio.h>

void main()
{
    
    FILE *fp1, *fp2;
    char ch;
    int pos;

    if ((fp1 = fopen("one.txt", "r")) == NULL)
    {
        printf("\nFile cannot be opened.");
        return;
    }
    else
    {
        printf("\nFile opened for copy...\n ");
    }
    fp2 = fopen("two.txt", "w");
    fseek(fp1, 0L, SEEK_END); 
    pos = ftell(fp1);
    fseek(fp1, 0L, SEEK_SET);
    while (pos--)
    {
        ch = fgetc(fp1);
        fputc(ch, fp2);
    }
    fclose(fp1);
    fclose(fp2);
}
